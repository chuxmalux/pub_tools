{ Windows 

	# generic pre-BOOT
	BIOS -> MBR -> Active Partition -> Bootloader 
		```
		B-M-A-B	
		being mindful allows balance !!!
		
		Boot, merge, augment, blast
		```
		
	# Broad 
		BIOS->OS Loader-> OS initialization
		B-O-O
	#  GOTO: Depth (kernel -> explorer init = OS initialization && ReadyBoot prefetcher)
		BIOS->OS Loader-> Kernel Init -> Session Init -> Winlogon Init -> Explorer Init. -> 'Post Boot' activity
			BIOS/ UEFI
			OS Loader: MBR -> PBR -> NTLDR 
					   BOOTMGR -> WINLOAD 
					   
		```
		B-O-K-S-W-E
		
		"Backstreets Of Kowloon, Such Wasted Excess"
			^ Just consider OS loader has 2 variants:
				- XP: BIOS -> OS Loader = (MBR->PBR->NTLDR) -> same 
				- Vista+: UEFI -> OS Lader = (winload.efi) -> same 
		```

	# XP/2003 in-depth Boot
	BIOS->MBR->PBS->NTLDR(boot.ini, ntdetect.com)->NTOSKRNL.exe(hal.dll)->smss.exe->winlogon(LSASS, SCM)
	B-M-P-N-N-S-W

	# 7, vista, 2008 in-depth Boot
	UEFI->Bootmgr(winresume.efi, BCD)->bootmgfw.efi->winload.efi->ntokrnl.exe(hal.dll)->smss.exe(win32K.sys,wininit.exe(LSASS))->winlogon.exe->logonui.exe->services.exe 
	U-B-W-N-S-W

	# LOGON process
	SAS -> winlogon/logon -> LSASS -> Driver Services -> Userinit -> Last Known Good -> Exit Userinit 
	S-W-L-D-U-L
}
